\section{Summer Hymns}
\begin{secrubric}
	The Summer Hymns are said from the Second Sunday after Trinity until the Sunday within 28 September - 4 October, exclusive.
\end{secrubric}

\subby{Sunday Mattins Office Hymn}
\fancyhead[RO,LE]{\textit{Summer Sunday}}
\fancyhead[RE,LO]{Mattins}


\gregorioscore{resources/gabc/Ferial/SundayMattinsSummer.gabc}

℣. The Lord is King, and hath put on glorious apparel.

℟. The Lord hath put on his apparel, and girded himself with strength.

\fancyhead[RE,LO]{Evensong}

℟. In thy sight as the incense.
